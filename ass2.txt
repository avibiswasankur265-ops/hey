#include <GL/glut.h>
#include <math.h>
void circle(float cx,float cy,float r)
{
    glBegin(GL_POLYGON);
    for(int i=0;i<100;i++)
    {
        float angle=2*3.1416*i/100;
        float x=r*cos(angle);
        float y=r*sin(angle);
        glVertex2f(cx+x,cy+y);
    }
    glEnd();
}

void drawFlower(float x,float y)
{
    glColor3f(0,0.6,0);
    glBegin(GL_LINES);
    glVertex2f(x,y);
    glVertex2f(x,y+0.03);
    glEnd();

    glColor3f(1,0,0);
    circle(x,y+0.035,0.01);

    glColor3f(1,1,0);
    circle(x,y+0.035,0.004);
}

void smoke(float x,float y)
{
    glColor3f(0.9,0.9,0.9);
    circle(x,y,0.015);
    circle(x+0.02,y+0.02,0.015);
    circle(x+0.04,y+0.04,0.015);
}
